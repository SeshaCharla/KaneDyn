\subsection{1(g)}
\itbf{Problem}: Referring to the Problem~$1(e)$, express the six quantities $x_i, y_i, z_i$ with $i=1,2$, eachas a function of a single quantitiy $q$ in such a way that the five constraint eqautions found previously are satisfied for all values of $q$. (Suggension: Let $q$ be the radian measure of the angle between $\pmb n_x$ and $\pmb p_2$.)

\itbf{Sol.}:

\begin{multicols}{2}
\begin{align*}
    x_1 &= -L_1 \cos q\\
    y_1 &= L_1 \sin q\\
    z_1 &=  0
\end{align*}

\begin{align*}
    x_2 &= L_2 \cos q\\
    y_2 &= -L_2 \sin q\\
    z_2 &=  0
\end{align*}
\end{multicols}
